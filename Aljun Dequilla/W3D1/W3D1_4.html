str.charAt(index)
Returns the the character at index of string
var anyString = "Brave"; anyString.charAt(0);
returns "B"


str.concat(string2, string3[, ..., stringN])
Combines the strings and return the result
var hello = "Hello"; console.log(hello.concat("Aljun"));
returns "HelloAljun"


str.indexOf(searchValue[, fromIndex])
Returns the index of the first search value
"Blue Whale".indexOf("Blue");
returns 0


str.lastIndexOf(searchValue[, fromIndex])
Returns the index of the last search value
"Blue Blue Blue".lastIndexOf("Blue");
returns 2


str.replace(regexp|substr, newSubStr|function)
Returns a new string with some or all matches of a pattern replaced by a replacement.
"Blue red Blue".replace("red","blue");
returns "Blue blue Blue"


str.toLowerCase()
Returns a string converted to lowercase
"Blue red Blue".toLowerCase();
returns "blue blue blue"
